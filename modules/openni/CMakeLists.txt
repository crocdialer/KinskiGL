##### OPENNI
if(WIN32)
    SET(OPENNI_INCLUDE_DIRS "")
    SET(OPENNI_LIB_DIR "")
    SET(OPENNI_LIBRARIES "OpenNI64")
else(WIN32)
    SET(OPENNI_INCLUDE_DIRS "/usr/include/ni")
    SET(OPENNI_LIB_DIR "/usr/lib")
    SET(OPENNI_LIBRARIES "OpenNI")
endif(WIN32)

set(MODULE_INCLUDES ${OPENNI_INCLUDE_DIRS} PARENT_SCOPE)
set(MODULE_LIBRARIES “${OPENNI_LIB_DIR}/OPENNI_LIBRARIES” PARENT_SCOPE)
